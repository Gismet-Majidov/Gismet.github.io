### Binary exponentiation
```cpp
#include<iostream>

#define ll long long

ll binpow(ll a, ll b)
{
    ll res = 1LL;
    while(b)
    {
        if(b&1)
            res = res * a;
        a = a * a;
        b>>=1;
    }
    return res;
}


int main()
{
    ll a = 3;
    ll b = 8;

    std::cout << binpow(a, b);

    return 0;
}
```
